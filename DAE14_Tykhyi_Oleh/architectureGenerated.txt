    %% Core Orchestrator
    class Game {
        -Map m_pMap
        -Player m_pPlayer
        -EnemyManager m_pEnemyManager
        -ParticleManager m_pParticleManager
        -CollisionManager m_pCollisionManager
        -CombatManager m_pCombatManager
        +Update()
        +Draw()
    }

    %% Main Subsystems
    class Map {
        -Vector2D m_Dimensions
        -TileType[][] m_Grid
        +Draw()
        +GetTileAt()
    }

    class EnemyManager {
        -vector<Enemy*> m_pEnemies
        -map<string, EnemyTemplate> m_Templates
        +Update()
        +Draw()
    }

    class ParticleManager {
        -vector<AttackParticle*> m_pParticles
        +Update()
        +Draw()
        +AddParticle()
    }

    %% Utility/Service Managers (Generally Statless or Global)
    class CollisionManager {
        +HandleMovement(Entity, Map)
    }
    class CombatManager {
        +ResolveCombat(Player, EnemyManager, ParticleManager)
    }

    %% Entity Hierarchy
    class Entity {
        <<Abstract>>
        -Sprite m_Sprite
        -Vector2D m_Position
        -Health m_Health
        +Update()
        +Draw()
    }

    class Player {
        +HandleInput()
    }

    class Enemy {
        <<Abstract>>
        -EnemyState m_CurrentState
        +UpdateIdle()
        +UpdatePatrol()
    }

    class Grunt {
    }

    class Gangster {
        -Sprite m_GunSprite
    }

    %% Particle System
    class AttackParticle {
        -Vector2D m_Velocity
        -Rect m_Hitbox
    }

    %% Ownership Relationships (Black Arrow = "Owns")
    Game *-- Map : Owns
    Game *-- Player : Owns
    Game *-- EnemyManager : Owns
    Game *-- ParticleManager : Owns
    Game *-- CollisionManager : Owns
    Game *-- CombatManager : Owns

    EnemyManager *-- Enemy : Owns (List)
    ParticleManager *-- AttackParticle : Owns (List)

    %% Inheritance Relationships (White Arrow = "Is A")
    Entity <|-- Player
    Entity <|-- Enemy
    Enemy <|-- Grunt
    Enemy <|-- Gangster

    %% Major Dependency/Flow Relationships (Dashed = "Uses/Calls")
    Game ..> CombatManager : Ticks Every Frame
    CombatManager ..> EnemyManager : Queries Enemies
    CombatManager ..> ParticleManager : Queries Particles
    Enemy ..> ParticleManager : SPWANS Particles
